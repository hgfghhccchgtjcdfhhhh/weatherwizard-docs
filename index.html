Quick Start

Get up and running with OpenWeather API integration in minutes.

JavaScript 
// Using OpenWeather API directly
const API_KEY = 'YOUR_OPENWEATHER_API_KEY';
const BASE_URL = 'https://api.openweathermap.org/data/2.5';

// Get current weather
async function getCurrentWeather(city) {
    const response = await fetch(
        `${BASE_URL}/weather?q=${city}&appid=${API_KEY}&units=metric`
    );
    const data = await response.json();
    return data;
}

// Usage
getCurrentWeather('New York').then(weather => {
    console.log(`Temperature: ${weather.main.temp}¬∞C`);
    console.log(`Condition: ${weather.weather[0].description}`);
});

string
City name, state code and country code divided by comma
appid
string
Your unique API key
units
string

Code Examples

JavaScript - Weather Dashboard 
const WeatherWizard = require('weatherwizard-sdk');
const client = new WeatherWizard(process.env.WEATHER_API_KEY);

async function createWeatherDashboard(cities) {
    const dashboard = {};
    
    for (const city of cities) {
        try {
            const weather = await client.getCurrentWeather({
                location: city,
                units: 'metric'
            });
            
            dashboard[city] = {
                temperature: weather.current.temperature,
                condition: weather.current.condition,
                humidity: weather.current.humidity
            };
        } catch (error) {
            console.error(`Failed to get weather for ${city}:`, error);
        }
    }
    
    return dashboard;
}

// Usage
const cities = ['London', 'Tokyo', 'New York', 'Sydney'];
createWeatherDashboard(cities).then(dashboard => {
    console.log('Weather Dashboard:', dashboard);
});
Python - Weather Alerts 
import weatherwizard
import asyncio

client = weatherwizard.Client(api_key="YOUR_API_KEY")

async def check_weather_alerts(locations):
    alerts = []
    
    for location in locations:
        weather = await client.get_current_weather(
            location=location,
            include_alerts=True
        )
        
        if weather.alerts:
            for alert in weather.alerts:
                alerts.append({
                    'location': location,
                    'type': alert.type,
                    'severity': alert.severity,
                    'description': alert.description
                })
    
    return alerts

# Check for alerts in multiple locations
locations = ["Miami, FL", "Houston, TX", "New Orleans, LA"]
alerts = await check_weather_alerts(locations)

for alert in alerts:
    print(f"‚ö†Ô∏è  {alert['location']}: {alert['description']}")
cURL - Basic Request 
# Get current weather
curl -X GET "https://weatherwizard-app.onrender.com/api/weather/current" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -G \
  -d "location=San Francisco, CA" \
  -d "units=metric"

# Get 7-day forecast
curl -X GET "https://weatherwizard-app.onrender.com/api/weather/forecast" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -G \
  -d "location=London, UK" \
  -d "days=7" \
  -d "units=metric"
